curl -X POST https://api.smslive247.com/v1/sms/send \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "to": "+234803XXXXXXX",
    "from": "YourBrand", 
    "message": "Hello from SMSLive247!"
  }'
const sms = require('@smslive247/sdk');

const client = new sms.Client('YOUR_API_KEY');

await client.messages.create({
  to: '+234803XXXXXXX',
  from: 'YourBrand',
  body: 'Hello from SMSLive247!'
});

client = smslive247.Client('YOUR_API_KEY')
